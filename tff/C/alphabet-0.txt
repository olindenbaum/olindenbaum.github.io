#include &lt;stdio.h&gt;
int main() {
    char c;
    printf("Enter a character: ");
    scanf("%c", &amp;c);

    if ((c &gt;= 'a' &amp;&amp; c &lt;= 'z') || (c &gt;= 'A' &amp;&amp; c &lt;= 'Z'))
        printf("%c is an alphabet.", c);
    else
        printf("%c is not an alphabet.", c);

    return 0;
}
